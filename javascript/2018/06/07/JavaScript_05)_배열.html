배열과 반복문의 조우

// 대문자로 바꿔주는 내장함수
var a = 'egoing'
a.toUpperCase();
// EGOING

// 배열의 개수를 나타내주는 내장함수
var a = ['egoing', 'k8805', 'sorialgi'];
alert(a.length);
// 3

function get_members() {
	return ['egoing', 'k8805', 'sorialgi'];
}

members = get_members();

for(var i = 0; i < members.length; i ++) {
	document.write(members[i].toUpperCase()+"<br />");
}

배열의 제어

// 배열의 뒤에 추가함
var li = ['a', 'b', 'c']
li.push('f');
alert(li);
// ['a', 'b', 'c', 'f']

// 배열과 배열을 합침
li = li.concat(['f', 'g']);
alert(li);
// ['a', 'b', 'c', 'f', 'f', 'g']


// 배열의 앞에 넣어줌
var li = ['a', 'b', 'c', 'd', 'e'];
li.unshift('z');
alert(li);
// ['z', 'a', 'b', 'c', 'd', 'e']


// 1번 인덱스부터 1개의 원소, 'b'를 삭제하고 그 위치에 'x', 'y'를 추가 함
var a = ['a', 'b', 'c'];
a.splice(1, 1, 'x', 'y');
alert(a);
// ['a', 'x', 'y', 'c']

var li = ['a', 'b', 'c', 'd', 'e'];

// 맨 앞의 원소를 제거하고 리턴함
li.shift();
alert(li);
// 'a'

// 맨 뒤의 원소를 제거하고 리턴함
li.pop();
// 'e'

// 정렬
var li = ['c', 'e', 'a', 'b', 'd'];
li.sort();
// ['a', 'b', 'c', 'd', 'e']

// 역순 정렬
li.reverse();
// ['e', 'd', 'c', 'b', 'a']

// sortfunction : 사용자 지정 정렬
