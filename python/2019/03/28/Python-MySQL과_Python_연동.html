CREATE DATABASE

QUERY = """
  CREATE DATABASE city_info_2;
"""
cur = db.cursor()

cur.execute(QUERY)

# 데이터 베이스 이동
QUERY = """
  USE city_info_2;
"""

cur = db.surcur()
cur.execute(QUERY)

# 테이블 생성
QUERY = """
  CREATE TABLE popular(
    rank int(4),
    city_name varchar(20),
    population int(10)
  );
"""

cur.execute(QUERY)

# 데이터 입력
QUERY = """
  INSERT INTO popular (rank, city_name, population)
  VALUES (1, 'seoul', 9900000),
  (2, 'busan', 3500000),
  (3, 'incheon', 2900000);
"""

cur.execute(QUERY)

# 데이터베이스와 cursor의 명령과 동기화시키는 명령
db.commit()
