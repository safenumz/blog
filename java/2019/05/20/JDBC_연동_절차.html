JDBC 연동 절차

  1. 해당 드라이버 메모리 로딩
  2. 연결객체 얻어오기
  3. sql 문장 만들기
  4. sql 전송 객체 얻어오기
    • Statement : 완벽한 sql
    • PreparedStatement : 미완성 sql
    • CllableStatement : pl-sql의 procedure/function 호출시
  5. sql 전송
    • INSERT/DELETE/UPDATE : int excuteUpdate()
    • SELECT : ResultSet executeQuery()
  6. 결과처리
  7. 닫기

외부에서 넘어오는 sql문은 auto commit 된다. 기본적으로 rollback, commit이 안된다. autocommit을 수동으로 풀어주면 가능

InfoVO.java

package jdbc.gui;

// VO(value of) = DTO = Record(X)
public class InfoVO {
    String name;
    String id;
    String tel;
    String gender;
    int age;
    String home;

    // constructor
    public InfoVO() {}

    public InfoVO(String name, String id, String tel, String gender, int age, String home) {
        this.name = name;
        this.id = id;
        this.tel = tel;
        this.gender = gender;
        this.age = age;
        this.home = home;
    }

    // getter, setter

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public String getId() {
        return id;
    }

    public void setId(String id) {
        this.id = id;
    }

    public String getTel() {
        return tel;
    }

    public void setTel(String tel) {
        this.tel = tel;
    }

    public String getGender() {
        return gender;
    }

    public void setGender(String gender) {
        this.gender = gender;
    }

    public int getAge() {
        return age;
    }

    public void setAge(int age) {
        this.age = age;
    }

    public String getHome() {
        return home;
    }

    public void setHome(String home) {
        this.home = home;
    }

    public String toString() {
        return name + "\t" + id + "\t" + tel + "\t" + gender + "\t" +
                age + "\t" + home + "\n";
    }
}

package jdbc.gui;

import java.sql.*;
import java.util.*;

public class InfoModelImpl {

    String driver = "oracle.jdbc.driver.OracleDriver";
    String url = "jdbc:oracle:thin:@safesoo7.iptime.org:1521:XE";
    String user = "oracle";
    String pass = "oracle";

    public InfoModelImpl() throws Exception {
        String driver = "oracle.jdbc.driver.OracleDriver";
        String url = "jdbc:oracle:thin:@safesoo7.iptime.org:1521:XE";
        String user = "oracle";
        String pass = "oracle";

        // 1. 드라이버를 메모리 로딩
        Class.forName(driver);
        // 2. 연결객체 얻어오기
    }

    public void insert(InfoVO vo) throws Exception {

        con = DriverManager.getConnection(url, user, pass);


        String sql = "INSERT INTO temp(name, id, tel, gender, age, home) " +
                " VALUES(?, ?, ?, ?, ?, ?)";

        PreparedStatement st = con.prepareStatement(sql);
        st.setString(1, vo.name);
        st.setString(2, vo.id);
        st.setString(3, vo.tel);
        st.setString(4, vo.gender);
        st.setInt(5, vo.age);
        st.setString(6, vo.home);

        System.out.println(sql);
        int result = st.executeUpdate();

        System.out.println(sql);
        st.close();
        con.close();

        /*
        2. 연결객체 얻어오기
        3. sql 문장 만들기
        String sql = "INSERT INTO 테이블명(컬럼명들) VALUES(?,?,?,?,?,?,?)";

        4. sql 전송 객체 얻어오기 - PreparedStatement : 미완성 sql

        5. sql 전송 : int executeUpdate()
        6. 결과처리
        7. 닫기
         */


    }

    public ArrayList<InfoVO> selectAllData() throws SQLException {
        Connection con = null;
        PreparedStatement st = null;
        ResultSet rs = null;

        try {
            // 2. 연결객체 얻어오기
            con = DriverManager.getConnection(url, user, pass);

            // 3. sql 문장 만들기
            String sql = "SELECT * FROM temp";

            // 4. sql 전송 객체 얻어오기 - PreparedStatement : 미완성 sql
            st = con.prepareStatement(sql);

            // 5. sql 전송 :
            rs = st.executeQuery();

            // 6. 결과처리
            // 제너릭스 -> 향상된 for문 가능
            ArrayList<InfoVO> list = new ArrayList<InfoVO>();

            while (rs.next()) {
                InfoVO vo = new InfoVO();
                vo.setName(rs.getString("name"));
                vo.setId(rs.getString("id"));
                vo.setTel(rs.getString("tel"));
                vo.setGender(rs.getString("gender"));
                vo.setAge(rs.getInt("age"));
                vo.setHome(rs.getString("home"));
                list.add(vo);
            }
            return list;
        } finally {
